**File da Creare/Modificare**:

```typescript
// vite.config.ts
import { defineConfig } from 'vite';
import react from '@vitejs/plugin-react';
import path from 'path';

export default defineConfig(({ mode }) => {
  const app = process.env.VITE_APP || 'formazione';
  
  return {
    plugins: [react()],
    resolve: {
      alias: {
        '@': path.resolve(__dirname, './src'),
        '@shared': path.resolve(__dirname, './src/shared'),
        '@formazione': path.resolve(__dirname, './src/apps/formazione'),
        '@medica': path.resolve(__dirname, './src/apps/medica'),
      },
    },
    build: {
      outDir: `dist/${app}`,
      rollupOptions: {
        input: {
          main: path.resolve(__dirname, app === 'medica' ? 'medica.html' : 'index.html'),
        },
      },
    },
    server: {
      port: app === 'medica' ? 5174 : 5173,
    },
  };
});
```

```json
// package.json - scripts aggiuntivi
{
  "scripts": {
    "dev": "VITE_APP=formazione vite",
    "dev:formazione": "VITE_APP=formazione vite",

    "build": "npm run build:formazione && npm run build:medica",
    "build:formazione": "VITE_APP=formazione vite build",
    "build:medica": "VITE_APP=medica vite build",
    "preview:formazione": "VITE_APP=formazione vite preview",
    "preview:medica": "VITE_APP=medica vite preview --port 5174"
  }
}
```
