```javascript
// backend/middleware/tenant.js - MODIFICHE

// Aggiungere mapping domini
const DOMAIN_APP_MAP = {
  'elementformazione.com': { tenantSlug: 'elementformazione', appType: 'FORMAZIONE' },
  'www.elementformazione.com': { tenantSlug: 'elementformazione', appType: 'FORMAZIONE' },
  'elementmedica.com': { tenantSlug: 'elementmedica', appType: 'MEDICA_PUBLIC' },
  'www.elementmedica.com': { tenantSlug: 'elementmedica', appType: 'MEDICA_PUBLIC' },
  'app.elementmedica.com': { tenantSlug: 'elementmedica', appType: 'MEDICA_PRIVATE' },
  'localhost:5173': { tenantSlug: 'development', appType: 'FORMAZIONE' },
  'localhost:5174': { tenantSlug: 'development', appType: 'MEDICA_PRIVATE' },
};

// App types enum
export const AppType = {
  FORMAZIONE: 'FORMAZIONE',
  MEDICA_PUBLIC: 'MEDICA_PUBLIC',
  MEDICA_PRIVATE: 'MEDICA_PRIVATE',
};

// Funzione helper per ottenere app type
export const getAppTypeFromDomain = (hostname) => {
  const mapping = DOMAIN_APP_MAP[hostname];
  return mapping?.appType || AppType.FORMAZIONE;
};

// Modificare middleware esistente
export const tenantMiddleware = async (req, res, next) => {
  try {
    const hostname = req.hostname || req.headers.host?.split(':')[0];
    
    // Ottieni mapping da dominio
    const mapping = DOMAIN_APP_MAP[hostname] || DOMAIN_APP_MAP[`${hostname}:${req.headers.host?.split(':')[1]}`];
    
    if (mapping) {
      req.tenantSlug = mapping.tenantSlug;
      req.appType = mapping.appType;
    } else {
      // Fallback a header o default
      req.tenantSlug = req.headers['x-tenant-slug'] || 'default';
      req.appType = req.headers['x-app-type'] || AppType.FORMAZIONE;
    }
    
    // Lookup tenant dal database
    const tenant = await prisma.tenant.findFirst({
      where: { 
        slug: req.tenantSlug,
        deletedAt: null 
      }
    });
    
    if (!tenant) {
      return res.status(404).json({ 
        success: false, 
        error: { code: 'TENANT_NOT_FOUND', message: 'Tenant non trovato' } 
      });
    }
    
    req.tenant = tenant;
    req.tenantId = tenant.id;
    
    // Log per debug
    logger.debug({ 
      hostname, 
      tenantSlug: req.tenantSlug, 
      appType: req.appType,
      tenantId: req.tenantId 
    }, 'Tenant resolved');
    
    next();
  } catch (error) {
    logger.error({ error }, 'Tenant middleware error');
    next(error);
  }
};
```

export const generateTokens = async (person, tenantId, appType) => {
  const accessTokenPayload = {
    personId: person.id,

    iat: Math.floor(Date.now() / 1000),

    process.env.JWT_REFRESH_SECRET,
    { expiresIn: process.env.JWT_REFRESH_EXPIRES_IN || '7d' }
  );
  
  return { accessToken, refreshToken };
};

// backend/middleware/auth.js - MODIFICHE

export const requireAuth = async (req, res, next) => {
  try {
    // ... existing token extraction ...
    
    const decoded = jwt.verify(token, process.env.JWT_SECRET);
    
    // NUOVO: Validazione appType
    if (req.appType && decoded.appType && req.appType !== decoded.appType) {
      // Permettere MEDICA_PRIVATE se logged in come MEDICA_PUBLIC (upgrade)
      const allowedUpgrades = {
        'MEDICA_PUBLIC': ['MEDICA_PRIVATE'],
      };
      
      const isAllowed = allowedUpgrades[decoded.appType]?.includes(req.appType);
      
      if (!isAllowed) {
        logger.warn({ 
          tokenAppType: decoded.appType, 
          requestAppType: req.appType,
          personId: decoded.personId 
        }, 'AppType mismatch');
        
        return res.status(403).json({
          success: false,
          error: { 
            code: 'APP_TYPE_MISMATCH', 
            message: 'Token non valido per questa applicazione. Effettua nuovamente il login.',
            requiredAppType: req.appType
          }
        });
      }
    }
    
    req.person = await prisma.person.findUnique({
      where: { id: decoded.personId },
      include: {
        roles: { include: { role: true } }
      }
    });
    
    req.personId = decoded.personId;
    req.tokenPayload = decoded;
    
    next();
  } catch (error) {
    // ... existing error handling ...
  }
};

# test-cross-domain-auth.sh

API_URL="http://localhost:4003"

echo "=== TEST CROSS-DOMAIN AUTH ==="

# T1: Login formazione
